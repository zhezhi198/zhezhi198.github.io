    auto check = [&](ll x, int n, ll m, const vector<ll> &a) -> bool
    {
        if (x == 0)
        {

            {
                if (a[i] <= a[i - 1])

                ll num = (d + x - 1) / x;
                sum += num;
                if (sum > m)
                    return false;
                cnt += num;
                lt = now + num * x;
            }
            else
            {
                lt = now;
            }
        }
        return sum <= m;
    };

    ll l = -1, r = 2e9 + 7;

    while (l + 1 < r)
    {
        ll mid = l + (r - l) / 2;
        if (check(mid, n, m, a))
        {
            r = mid;
        }
        else
        {
            l = mid;
        }
    }
    if (r == 2e9 + 7)
    {
        cout << -1 << endl;
    }
    else
    {
        cout << r << endl;
    }
