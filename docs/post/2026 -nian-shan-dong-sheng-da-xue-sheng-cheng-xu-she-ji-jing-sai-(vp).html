void solve()
{
    ll n;
    cin >> n;
    vector<PLL> p(n + 1);
    for (int i = 1; i <= n; i++)
    {
        ll a, b;
        cin >> a >> b;
        p[i] = {a, b};
    }
    sort(p.begin() + 1, p.end(), [](const PLL &x, PLL &y)
         {
        if(x.fi==y.fi) return x.se>y.se;
        return x.fi<y.fi; });
    vll k, nk;
    ll lt = -1, ans = 0;
    for (int i = 1; i <= n;)
    {
        int j = i;
        while (j <= n && p[i].fi == p[j].fi)
        {
            j++;
        }
        if (p[i].fi != lt + 1)
        {
            k.clear();
        }
        nk.clear();
        for (int idx = i; idx < j; idx++)
        {
            int cnt = idx - i;
            ll now = 1;
            if (cnt < k.size())
            {
                now = k[cnt] + 1;
            }
            ans += now * p[idx].se;
            nk.push_back(now);
        }
        lt = p[i].fi;
        i = j;
        k = nk;
    }
    cout << ans << endl;
}

{

    {
        solve();
    }
    return 0;
}

{
    int n, m;
    cin >> n >> m;
    int sum = n * m;
    if (sum <= 4)
    {
        cout << "No" << endl;
        return;
    }
    if (sum == 6)
    {
        cout << "Yes" << endl;
        if (n == 1 && m == 6)
        {
            cout << "0 2 4 1 5 3" << endl;
        }
        if (n == 6 && m == 1)
        {
            cout << "0" << endl;
            cout << "2" << endl;
            cout << "4" << endl;
            cout << "1" << endl;
            cout << "3" << endl;
            cout << "5" << endl;
        }
        if (n == 2 && m == 3)
        {
            cout << "5 3 1" << endl;
            cout << "2 0 4" << endl;
        }
        if (n == 3 && m == 2)
        {
            cout << "1 4" << endl;
            cout << "3 0" << endl;
            cout << "5 2" << endl;
        }
        return;
    }
    ll k = -1;
    for (int i = 2; i < sum - 1; i++)
    {
        if (gcd(i, sum) != 1)
            continue;
        if (n > 1)
        {
            ll d = i * m % sum;
            if (d == 1 || d == sum - 1)
                continue;
        }
        k = i;
        break;
    }
    if (k == -1)
    {
        cout << "No" << endl;
        return;
    }
    else
    {
        cout << "Yes" << endl;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < m; j++)
            {
                cout << (i * m + j) * k % sum << " ";
            }
            cout << endl;
        }
    }
}

        else if (cku && !ckv)
        {
            mn[u] = min(mn[u], w);
        }
    }
    if (n == k)
    {
        cout << -1 << endl;
        return;
    }
    sort(tr + 1, tr + idx + 1, cmp);
    for (int i = 1; i <= n; i++)
        p[i] = i;
    ll ans = 0;
    ll cnt = 0;
    for (int i = 1; i <= idx; i++)
    {
        auto [u, v, w] = tr[i];
        int pu = find(u), pv = find(v);
        if (pu != pv)
        {
            merge(u, v);
            ans += w;
            cnt++;
        }
    }
    if (cnt < n - k - 1)
    {
        cout << -1 << endl;
        return;
    }
    for (int i = 1; i <= k; i++)
    {
        if (mn[a[i]] == inf)
        {
            cout << -1 << endl;
            return;
        }
        else
        {
            ans += mn[a[i]];
        }
    }
    cout << ans << endl;
}

void solve()
{
    ll n;
    cin >> n;
    vector<PLL> p(n + 1);
    for (int i = 1; i <= n; i++)
    {
        ll l, r;
        cin >> l >> r;
        p[i] = {l, r};
    }
    sort(p.begin() + 1, p.end());
    auto ck = [&](ll x) -> ll
    {
        ll res = 0;
        for (int i = 1; i <= n; i++)
        {
            auto [l, r] = p[i];
            if (x >= l && x <= r)
                continue;
            else if (x < l)
                res += (l - x) * (l - x);
            else if (x > r)
                res += (x - r) * (x - r);
        }
        return res;
    };
    ll l = -1, r = 1000000 + 1;
    while (l + 1 < r)
    {
        ll mid = (l + r) >> 1;
        if (ck(mid) <= ck(mid + 1))
            r = mid;
        else
            l = mid;
    }
    cout << ck(r) << endl;
}
