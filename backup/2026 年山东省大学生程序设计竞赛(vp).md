        }
        else if (cku && !ckv)
        {
            mn[u] = min(mn[u], w);
        }
    }
    if (n == k)
    {
        cout << -1 << endl;
        return;
    }
    sort(tr + 1, tr + idx + 1, cmp);
    for (int i = 1; i <= n; i++)
        p[i] = i;
    ll ans = 0;
    ll cnt = 0;
    for (int i = 1; i <= idx; i++)
    {
        auto [u, v, w] = tr[i];
        int pu = find(u), pv = find(v);
        if (pu != pv)
        {
            merge(u, v);
            ans += w;
            cnt++;
        }
    }
    if (cnt < n - k - 1)
    {
        cout << -1 << endl;
        return;
    }
    for (int i = 1; i <= k; i++)
    {
        if (mn[a[i]] == inf)
        {
            cout << -1 << endl;
            return;
        }
        else
        {
            ans += mn[a[i]];
        }
    }
    cout << ans << endl;
}

    {
        solve();
    }
    return 0;
}
